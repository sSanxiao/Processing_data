import pandas as pd

# 读取tsv文件
mat = pd.read_csv('D:/Mat.tsv', sep='\t', index_col=0)

# 将tsv文件分为mat_train和mat_test
mid_col = mat.shape[1] // 2
mat_train = mat.iloc[:, :mid_col]
mat_test = mat.iloc[:, mid_col:]

# 保存mat_train和mat_test为tsv文件
mat_train.to_csv('D:/mat_train.tsv', sep='\t')
mat_test.to_csv('D:/mat_test.tsv', sep='\t')

# 读取csv文件
correct_dp = pd.read_csv('D:/CorrectDP.csv', index_col=0)

# 将csv文件分为correct_dp_train和correct_dp_test
mid_row = correct_dp.shape[0] // 2
correct_dp_train = correct_dp.iloc[:mid_row, :]
correct_dp_test = correct_dp.iloc[mid_row:, :]

# 将correct_dp_train和correct_dp_test的行名与mat_train和mat_test的列名对应
correct_dp_train = correct_dp_train.loc[mat_train.columns]
correct_dp_test = correct_dp_test.loc[mat_test.columns]

# 保存correct_dp_train和correct_dp_test为csv文件
correct_dp_train.to_csv('D:/CorrectDP_train.csv')
correct_dp_test.to_csv('D:/CorrectDP_test.csv')
